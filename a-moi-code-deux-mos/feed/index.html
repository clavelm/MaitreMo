function wp_touch_get_comment_count() {
global $wpdb;
global $post;
$result = $wpdb->get_row( $wpdb->prepare( "SELECT count(*) as c FROM {$wpdb->comments} WHERE comment_type = ' AND comment_approved = 1 AND comment_post_ID = %d", $post->ID ) );
if ( $result ) {
return $result->c;
} else {
return 0;
}
}
